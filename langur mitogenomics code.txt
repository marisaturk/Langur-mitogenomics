###Deduplicating reads and calculating mitogenome coverage###

#!/bin/bash
#
#SBATCH --job-name=samtools-batch
#SBATCH --ntasks-per-node=1
#SBATCH --nodes=1
#SBATCH --time=04:00:00
#SBATCH --mail-type=BEGIN,END
#SBATCH --mail-user=marisa.turk@stonybrook.edu
#SBATCH -p short-40core
module load shared
cd $SLURM_SUBMIT_DIR
module load hts/1.0

for i in *sam; do
echo $i

samtools view -S -b -q 30 -F 4 $i > ${i%.*}.bam

samtools sort ${i%.*}.bam -o ${i%.*}.srt.bam

samtools rmdup -s ${i%.*}.srt.bam ${i%.*}.dup.bam

samtools mpileup -BQ0 -d10000000 -f ./NC_019582.1.fasta ./${i%.*}.dup.bam | awk '{ count++ ; SUM += $4 } END { print "Total ${i%.*}: " SUM "\t" "Nucleotides: " count "\t" "Average_coverage: " SUM/count }'

done 

---

###Samtools for Variant Calling###
#Variant calling
#code from https://samtools.github.io/bcftools/howtos/variant-calling.html

bcftools mpileup -Ou -f NC_019582.1.fasta LIB003224.dup.bam | bcftools call -mv -Ob -o LIB003224calls.bcf

for i in *.dup.bam; do
bcftools mpileup -Ou -f NC_019582.1.fasta $i | bcftools call -mv -Ob -o ${i%.*.*}_calls.bcf
done

for i in *bcf; do
bcftools index $i
done

---

#Filtering
#Min qual score of 20, min read depth of 4

for i in *bcf; do
bcftools filter -i'QUAL>20' $i | bcftools filter -i'DP>3' $i -o ${i%.*}_filtered.bcf
done

for i in *filtered.bcf; do
bcftools index $i
done

---

###Creating consensus sequences###
#workflow from https://samtools.github.io/bcftools/howtos/consensus-sequence.html

#Normalize indels
for i in *filtered.bcf; do
bcftools norm -f NC_019582.1.fasta $i -Ob -o ${i%.*}.norm.bcf
done

#Filter adjacent indels within 5bp
for i in *norm.bcf; do
bcftools filter --IndelGap 5 $i -Ob -o ${i%.*.*}.norm.flt-indels.bcf
done

#Apply variants to create consensus sequence
for i in *filtered.bcf; do
cat NC_019582.1.fasta | bcftools consensus $i > ${i%_*_*}_consensus.fa
done

#Output IUPAC ambiguity codes based on REF+ALT columns (regardless of genotype)
for i in *filtered.bcf; do
cat NC_019582.1.fasta | bcftools consensus --iupac-codes $i > ${i%_*_*}_consensus.fa
done

#Output IUPAC ambiguity codes based on sample genotypes
for i in *filtered.bcf; do
cat NC_019582.1.fasta | bcftools consensus --haplotype I $i > ${i%_*_*}_consensus.fa
done

---

###Clustal Omega for multiple alignment###
clustal_omega -i subset_90_consensus.fa --outfmt fa > subset_90_consensus.clustalo.fa

###RAxML for ML tree construction###
raxmlHPC-SSE3 -s subset_90_consensus.clustalo.fa -n FullRAxMLOutput -m GTRGAMMA -p 222 -N 1000 -f a -x 222
